100  private void CombatThread()
101  {
102  WinningPartyId = null;
103  _currentGameState.State = CombatState.Active;
104 
105  uint winningParty = 0u;
106  while (!GetWinningPartyId(out winningParty))
107  {
108  // Get Active Entity
109  IInitiativeActor activeEntity = _initiativeQueue.GetNext();
110 
111  // Update GameState with ActiveEntity
112  _previousGameState = (GameState.GameState) _currentGameState.Copy();
113  _currentGameState.ActiveActor = activeEntity;
114  _currentGameState.Id = GameState.GameState.NextId;
115  SendOutgoingGameState();
116 
117  _previousGameState = (GameState.GameState) _currentGameState.Copy();
118  _currentGameState.Id = GameState.GameState.NextId;
119 
120  // Prompt Action
121  if (activeEntity.IsAlive())
122  {
123  // Get Controller and Entity Actions for Active Entity
124  ICharacterController controller = _controllerByPartyId[activeEntity.Party];
125  Dictionary<uint, IAction> actionsForEntity = activeEntity
126  .GetAllActions(_allTargets)
127  .ToDictionary(x => x.Id);
128 
129  // Add Standard Entity Actions
130  foreach (IAction standardAction in _standardActionGenerator.GetActions(activeEntity))
131  {
132  actionsForEntity[standardAction.Id] = standardAction;
133  }
134 
135  // Wait for valid response
136  IAction selectedAction;
137  do
138  {
139  // Send Actions to Controller
140  _pendingSelectedActionId = null;
141  controller.SelectAction(activeEntity, actionsForEntity, OnActionSelected);
142 
143  // Wait for Controller response
144  do
145  {
146  Thread.Sleep(250);
147  } while (!_pendingSelectedActionId.HasValue);
148 
149  // Validate response
150  actionsForEntity.TryGetValue(_pendingSelectedActionId.Value, out selectedAction);
151 
152  } while (selectedAction == null || !selectedAction.Available);
153 
154  // Apply effects of Action
155  _initiativeQueue.Update(activeEntity, -selectedAction.Speed);
156 
157  selectedAction.Cost.Pay(selectedAction.Source, selectedAction.ActionSource);
158  selectedAction.Effect.Apply(selectedAction.Source, selectedAction.Targets);
159  }
160  else
161  {
162  _initiativeQueue.Update(activeEntity, -_targetInitiative);
163  }
164 
165  // Update GameState with Effects of Action
166  _currentGameState.ActiveActor = null;
167  SendOutgoingGameState();
168 
169  // Rest for a bit
170  Thread.Sleep(1000);
171  }
172 
173  _previousGameState = (GameState.GameState) _currentGameState.Copy();
174  _currentGameState.Id = GameState.GameState.NextId;
175  _currentGameState.State = CombatState.Completed;
176 
177  WinningPartyId = winningParty;
178  SendOutgoingGameState();
179 
180  CombatComplete.Fire(winningParty);
181  }
